import { useState } from "react";

function UseStateHook(){
    // console.log( useState() );
    let [power, stateAccess] = useState(0);
    
    return <div>
                <h2>Hooks</h2>
                <hr/>
                <button onClick={ ()=>stateAccess(power+1) }>Incease Power</button>
                <h3>Power : { power }</h3>
           </div>

}

export default UseStateHook;
